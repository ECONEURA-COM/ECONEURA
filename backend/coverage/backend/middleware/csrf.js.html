/**
 * CSRF Protection
 * Implementación moderna usando double-submit cookie pattern
 * (csurf está deprecated, esta es la alternativa recomendada)
 */

const crypto = require("crypto");
const { logSecurityEvent } = require("../logger");

const CSRF_TOKEN_LENGTH = 32;
const CSRF_COOKIE_NAME = "csrf-token";
const CSRF_HEADER_NAME = "x-csrf-token";
const ENVIRONMENT = process.env.NODE_ENV || "development";

/**
 * Generate CSRF token
 */
function generateCsrfToken() {
  return crypto.randomBytes(CSRF_TOKEN_LENGTH).toString("hex");
}

/**
 * CSRF Protection Middleware
 * Double-submit cookie pattern
 */
function csrfProtection(options = {}) {
  const {
    ignoreMethods = ["GET", "HEAD", "OPTIONS"],
    cookieOptions = {
      httpOnly: true,
      secure: ENVIRONMENT === "production",
      sameSite: "strict",
      maxAge: 3600000, // 1 hora
    },
  } = options;
  
  return (req, res, next) => {
    const method = req.method.toUpperCase();
    
    // Skip CSRF para métodos seguros
    if (ignoreMethods.includes(method)) {
      // Pero establecer token en cookie para uso futuro
      if (!req.cookies || !req.cookies[CSRF_COOKIE_NAME]) {
        const token = generateCsrfToken();
        res.cookie(CSRF_COOKIE_NAME, token, cookieOptions);
        req.csrfToken = () => token;
      } else {
        req.csrfToken = () => req.cookies[CSRF_COOKIE_NAME];
      }
      return next();
    }
    
    // Verificar token para métodos no seguros (POST, PUT, DELETE, etc.)
    const cookieToken = req.cookies?.[CSRF_COOKIE_NAME];
    const headerToken = req.headers[CSRF_HEADER_NAME] || req.body?._csrf;
    
    if (!cookieToken || !headerToken) {
      logSecurityEvent("CSRF: Missing token", {
        method,
        path: req.path,
        ip: req.ip,
        hasCookie: !!cookieToken,
        hasHeader: !!headerToken,
      });
      
      return res.status(403).json({
        error: "Forbidden",
        message: "CSRF token missing",
      });
    }
    
    // Comparación timing-safe
    if (!crypto.timingSafeEqual(Buffer.from(cookieToken), Buffer.from(headerToken))) {
      logSecurityEvent("CSRF: Invalid token", {
        method,
        path: req.path,
        ip: req.ip,
      });
      
      return res.status(403).json({
        error: "Forbidden",
        message: "CSRF token mismatch",
      });
    }
    
    // Token válido
    req.csrfToken = () => cookieToken;
    next();
  };
}

/**
 * Middleware para agregar CSRF token a response
 * Útil para SPA que necesitan el token
 */
function csrfTokenProvider() {
  return (req, res, next) => {
    // Agregar método helper para obtener token
    res.locals.csrfToken = req.csrfToken ? req.csrfToken() : null;
    next();
  };
}

/**
 * Endpoint para obtener CSRF token
 * GET /api/csrf-token
 */
function getCsrfToken(req, res) {
  const token = req.csrfToken ? req.csrfToken() : generateCsrfToken();
  
  res.json({
    csrfToken: token,
  });
}

module.exports = {
  csrfProtection,
  csrfTokenProvider,
  getCsrfToken,
  generateCsrfToken,
};
