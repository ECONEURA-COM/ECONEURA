/**
 * Rate Limiting Middleware
 *
 * Protección contra abuso con límites configurables:
 * - Por IP: 100 requests / 15 min (default)
 * - Por API key: 1000 requests / 15 min
 * - Sliding window para precisión
 *
 * Backends soportados:
 * - Memoria (default): Rápido, desarrollo, no distribuido
 * - Upstash Redis (opcional): Producción, distribuido, persistente
 */

// ===== BACKEND: IN-MEMORY =====
class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanup();
  }

  /**
   * Incrementar contador para una clave
   * @param {string} key - Identificador (IP o API key)
   * @param {number} windowMs - Ventana en ms
   * @returns {Promise<{count: number, resetAt: number}>}
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const data = this.store.get(key) || { count: 0, resetAt: now + windowMs };

    // Resetear si la ventana expiró
    if (now >= data.resetAt) {
      data.count = 0;
      data.resetAt = now + windowMs;
    }

    data.count++;
    this.store.set(key, data);

    return {
      count: data.count,
      resetAt: data.resetAt
    };
  }

  /**
   * Limpiar entradas expiradas cada 5 minutos
   */
  cleanup() {
    setInterval(() => {
      const now = Date.now();
      for (const [key, data] of this.store.entries()) {
        if (now >= data.resetAt + 60000) { // +1 min grace
          this.store.delete(key);
        }
      }
    }, 5 * 60 * 1000);
  }
}

let UpstashStore = null;
if (process.env.UPSTASH_REDIS_REST_URL && process.env.UPSTASH_REDIS_REST_TOKEN) {
  try {
    const { Redis } = require('@upstash/redis');

    class UpstashStoreImpl {
      constructor() {
        this.redis = new Redis({
          url: process.env.UPSTASH_REDIS_REST_URL,
          token: process.env.UPSTASH_REDIS_REST_TOKEN,
        });
      }

      async increment(key, windowMs) {
        const now = Date.now();
        const windowKey = `ratelimit:${key}:${Math.floor(now / windowMs)}`;

        // Incrementar en Redis
        const count = await this.redis.incr(windowKey);

        // TTL para auto-cleanup
        if (count === 1) {
          await this.redis.expire(windowKey, Math.ceil(windowMs / 1000) + 60);
        }

        return {
          count,
          resetAt: Math.ceil(now / windowMs) * windowMs + windowMs
        };
      }
    }

    UpstashStore = UpstashStoreImpl;
    // Rate limiting: Upstash Redis enabled (silent)
  } catch (err) {
    // Upstash Redis not available, using in-memory fallback (silent)
  }
}

/**
 * Rate limiting middleware
 * @param {object} options - Configuración
 * @param {number} options.windowMs - Ventana de tiempo (default: 15 min)
 * @param {number} options.max - Máximo de requests (default: 100)
 * @param {string} options.message - Mensaje de error (default: "Too many requests")
 * @param {function} options.keyGenerator - Función para generar key (default: IP)
 */
function rateLimit(options = {}) {
  const {
    windowMs = 15 * 60 * 1000, // 15 minutos
    max = 100,
    message = 'Too many requests, please try again later',
    keyGenerator = (req) => {
      // Prioridad: API key > IP forwarded > IP directa
      return req.headers['x-api-key'] ||
             req.headers['x-forwarded-for']?.split(',')[0].trim() ||
             req.ip ||
             req.connection.remoteAddress;
    },
    skip = () => false
  } = options;

  return async (req, res, next) => {
    // Skip si la función lo indica (ej: health checks)
    if (skip(req)) {
      return next();
    }

    try {
      const key = keyGenerator(req);

      if (!key) {
        // No key generated, allowing request (silent)
        return next();
      }

      const { count, resetAt } = await store.increment(key, windowMs);

      // Headers estándar de rate limiting
      res.setHeader('X-RateLimit-Limit', max);
      res.setHeader('X-RateLimit-Remaining', Math.max(0, max - count));
      res.setHeader('X-RateLimit-Reset', Math.ceil(resetAt / 1000));

      if (count > max) {
        const retryAfter = Math.ceil((resetAt - Date.now()) / 1000);
        res.setHeader('Retry-After', retryAfter);

        return res.status(429).json({
          error: message,
          retryAfter,
          limit: max,
          window: windowMs / 1000
        });
      }

      next();
    } catch (err) {
      // Si falla rate limiting, no bloquear (silent, fail open)
      next();
    }
  };
}

/**
 * Rate limit estricto para APIs públicas
 * 60 requests / 15 min por IP
 */
const strictRateLimit = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 60,
  message: 'API rate limit exceeded. Please try again in 15 minutes.'
});

/**
 * Rate limit moderado para APIs autenticadas
 * 1000 requests / 15 min por API key
 */
const apiKeyRateLimit = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 1000,
  keyGenerator: (req) => req.headers['x-api-key'] || req.ip,
  message: 'API key rate limit exceeded. Please try again later.'
});

/**
 * Rate limit ligero para desarrollo
 * 500 requests / 15 min
 */
const devRateLimit = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 500,
  skip: (req) => process.env.NODE_ENV === 'development' && req.ip === '::1'
});
