const promptsCache = {};

function cargarPrompt(agentId) {
  // Mapeo de agentId â†’ archivo prompt
  const promptMap = {
    'a-ceo-01': 'neura-ceo-v1.2.md',
    'a-ia-01': 'neura-ia-v1.2.md',
    'a-cso-01': 'neura-cso-v1.2.md',
    'a-cto-01': 'neura-cto-v1.2.md',
    'a-ciso-01': 'neura-ciso-v1.2.md',
    'a-coo-01': 'neura-coo-v1.2.md',
    'a-chro-01': 'neura-chro-v1.2.md',
    'a-mkt-01': 'neura-mkt-v1.2.md',
    'a-cfo-01': 'neura-cfo-v1.2.md',
    'a-cdo-01': 'neura-cdo-v1.2.md'
  };

  const fileName = promptMap[agentId];
  if (!fileName) {
    return 'Eres un asistente ejecutivo de ECONEURA. Responde de forma profesional y concisa.';
  }

  // Cache para evitar leer archivo cada vez
  if (promptsCache[agentId]) {
    return promptsCache[agentId];
  }

  try {
    const filePath = path.join(promptsDir, fileName);
    const content = fs.readFileSync(filePath, 'utf8');
    promptsCache[agentId] = content;
    log.info('Prompt loaded', { agentId, fileName });
    return content;
  } catch (err) {
    log.error('Failed to load prompt', { agentId, fileName, error: err.message });
    return 'Eres un asistente ejecutivo de ECONEURA. Responde de forma profesional y concisa.';
  }
}

// Cliente OpenAI (sin dependencias)
function llamarOpenAI(messages) {
  return new Promise((resolve, reject) => {
    if (!OPENAI_KEY) {
      log.error('OPENAI_API_KEY not loaded from .env');
      return resolve('');
    }

    const payload = {
      model: 'gpt-4o-mini',
      messages: messages,
      max_tokens: 500,
      temperature: 0.7
    };
    const data = JSON.stringify(payload);
    const dataBuffer = Buffer.from(data, 'utf8');

    const options = {
      hostname: 'api.openai.com',
      port: 443,
      path: '/v1/chat/completions',
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${OPENAI_KEY}`,
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(dataBuffer)
      }
    };

    const req = https.request(options, (res) => {
      let body = '';
      res.setEncoding('utf8');
      res.on('data', (chunk) => body += chunk);
      res.on('end', () => {
        try {
          const json = JSON.parse(body);
          const texto = json.choices?.[0]?.message?.content || '';
          if (texto) {
            log.info('OpenAI success', { length: texto.length });
            resolve(texto.trim());
          } else {
            log.error('OpenAI API error', json.error);
            resolve('');
          }
        } catch (err) {
          log.error('OpenAI parse error', err);
          resolve('');
        }
      });
    });

    req.on('error', (err) => {
      log.error('OpenAI request error', err);
      resolve('');
    });

    req.write(dataBuffer);
    req.end();
  });
}

app.post('/api/invoke/:id', optionalAuth, async (req, res) => {
  const { input } = req.body;
  const { id: agentId } = req.params;

  if (!agentId || !agentId.includes('-')) {
    return res.json({ output: 'Fallback: Invalid agentId', provider: 'error' });
  }
  const dept = agentId.split('-')[1] || 'default';

  if (!input) {
    return res.status(400).json({ error: 'Input requerido' });
  }

  log.info(`Chat request`, { agentId, input: input.substring(0, 50) });

  try {
    // Cargar Prompt Maestro v1.2 personalizado para este NEURA
    const systemContent = cargarPrompt(agentId);
    log.debug('System prompt loaded', { agentId, length: systemContent.length });

    const messages = [
      { role: 'system', content: systemContent },
      { role: 'user', content: input }
    ];

    const startTime = Date.now();

    log.debug('OpenAI request', { agentId, keyLoaded: !!OPENAI_KEY });

    let output = await llamarOpenAI(messages);
    if (!output) output = 'Error: OpenAI failed - check key';

    const duration = Date.now() - startTime;

    log.chat(agentId, input, output, duration);

    // Guardar en DB si hay usuario autenticado
    if (req.user && process.env.DATABASE_URL) {
      try {
        const { neon } = require('@neondatabase/serverless');
        const sql = neon(process.env.DATABASE_URL);
        await sql`
          INSERT INTO chats (user_id, neura_id, input, output, provider, model, duration, created_at)
          VALUES (${req.user.id}, ${agentId}, ${input}, ${output}, 'openai', 'gpt-4o-mini', ${duration}, NOW())
        `;
      } catch (dbError) {
        // Log error pero no bloquear respuesta
        log.error('Failed to save chat to DB', dbError);
      }
    }

    const response = {
      output: output,
      provider: 'openai',
      model: 'gpt-4o-mini',
      agentId
    };

    res.json(response);

  } catch (error) {
    log.error('Chat error', error, { agentId, input: input?.substring(0, 50) });
    res.status(500).json({
      error: error.message,
      provider: 'error'
    });
  }
});

// Start server with Key Vault initialization
async function startServer() {
  try {
    // Initialize secrets from Key Vault
    console.log('[Startup] Initializing Azure Key Vault...');
    const kvInitialized = await initializeSecrets();
    
    if (kvInitialized || process.env.NODE_ENV !== 'production') {
      // Load OPENAI_KEY from Key Vault or fallback
      try {
        OPENAI_KEY = await getSecret('OPENAI-API-KEY');
        console.log('[Startup] OpenAI key loaded from Key Vault');
      } catch (error) {
        OPENAI_KEY = process.env.OPENAI_API_KEY;
        console.warn('[Startup] Using OPENAI_API_KEY from environment (fallback)');
      }
    }
    
    // Start Express server
    app.listen(PORT, () => {
      log.info('Server started', {
        port: PORT,
        provider: 'openai',
        model: 'gpt-4o-mini',
        environment: process.env.NODE_ENV || 'development',
        keyVault: kvInitialized ? 'enabled' : 'fallback'
      });

      const banner = [
        '',
        '='.repeat(55),
        '  ECONEURA - Backend Ready v2.0',
        '='.repeat(55),
        `  Server:   http://localhost:${PORT}`,
        '  Provider: OpenAI (gpt-4o-mini)',
        '  Logging:  Winston',
        `  Key Vault: ${kvInitialized ? "OK" : "Fallback"}`,
        '  Status:   Ready',
        '='.repeat(55),
        ''
      ].join('\n');

      if (process.env.NODE_ENV !== 'production') {
        // eslint-disable-line no-console
        console.log(banner);
      }
    });
    
  } catch (error) {
    console.error('[Startup] Failed to start:', error);
    process.exit(1);
  }
}
